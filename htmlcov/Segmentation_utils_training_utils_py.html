4def dice_coef(y_true, y_pred, smooth=1): 

5 """ 

6 Dice = (2*|X & Y|)/ (|X|+ |Y|) 

7 = 2*sum(|A*B|)/(sum(A^2)+sum(B^2)) 

8 ref: https://arxiv.org/pdf/1606.04797v1.pdf 

9 """ 

10 intersection = K.sum(K.abs(y_true * y_pred), axis=-1) 

11 return (2. * intersection + smooth) / (K.sum(K.square(y_true),-1) + K.sum(K.square(y_pred),-1) + smooth)

16def cross_entropy_loss(labels, logits, n_classes, loss_mask=None, data_format='channels_last', one_hot_labels=True, name='ce_loss'): 

17 """ 

18 Cross-entropy loss. 

19 :param labels: 4D tensor 

20 :param logits: 4D tensor 

21 :param n_classes: integer for number of classes 

22 :param loss_mask: binary 4D tensor, pixels to mask should be marked by 1s 

23 :param data_format: string 

24 :param one_hot_labels: bool, indicator for whether labels are to be expected in one-hot representation 

25 :param name: string 

26 :return: dict of (pixel-wise) mean and sum of cross-entropy loss 

27 """ 

28 # permute class channels into last axis 

29 if data_format == 'channels_first': 

30 labels = tf.transpose(labels, [0,2,3,1]) 

31 logits = tf.transpose(logits, [0,2,3,1]) 

32 

33 batch_size = tf.cast(tf.shape(labels)[0], tf.float32) 

34 

35 if one_hot_labels: 

36 flat_labels = tf.reshape(labels, [-1, n_classes]) 

37 else: 

38 flat_labels = tf.reshape(labels, [-1]) 

39 flat_labels = tf.one_hot(indices=flat_labels, depth=n_classes, axis=-1) 

40 flat_logits = tf.reshape(logits, [-1, n_classes]) 

41 

42 # do not compute gradients w.r.t the labels 

43 flat_labels = tf.stop_gradient(flat_labels) 

44 

45 ce_per_pixel = tf.nn.softmax_cross_entropy_with_logits(labels=flat_labels, logits=flat_logits) 

46 

47 # optional element-wise masking with binary loss mask 

48 if loss_mask is None: 

49 ce_sum = tf.reduce_sum(ce_per_pixel) / batch_size 

50 ce_mean = tf.reduce_mean(ce_per_pixel) 

51 else: 

52 loss_mask_flat = tf.reshape(loss_mask, [-1,]) 

53 loss_mask_flat = (1. - tf.cast(loss_mask_flat, tf.float32)) 

54 ce_sum = tf.reduce_sum(loss_mask_flat * ce_per_pixel) / batch_size 

55 n_valid_pixels = tf.reduce_sum(loss_mask_flat) 

56 ce_mean = tf.reduce_sum(loss_mask_flat * ce_per_pixel) / n_valid_pixels 

57 

58 return {'sum': ce_sum, 'mean': ce_mean}
